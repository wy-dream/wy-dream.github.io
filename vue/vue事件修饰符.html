# vue事件修饰符

a .stop → 阻止事件冒泡，不添加stop的时候点击box3时，会依次弹出box3>box2>box1,添加stop之后box3弹出来以后就不会继续弹了，如下：

  <div id="app">
      <div class="box1" v-on:click="say1">
          加了self以后只能点自己才能触发
          <div class="box2" v-on:click="say2">
              <div class="box3" v-on:click.stop="say3">谁弹出来</div>
          </div>
      </div>
  </div>

  <script>
      new Vue({
          el: "#app",
          methods: {
              say1() {
                  alert("box1")
              },
              say2() {
                  alert("box2")
              },
              say3() {
                  alert("box3")
              },
          }
      })
  </script>

b .capture → 事件捕获，不添加capture的时候点击box3时，会依次弹出box3>box2>box1,添加capture之后点击box3时，会弹出box1>box3>box2，如下：

  <div id="app">
      <div class="box1" v-on:click.capture="say1">
          加了self以后只能点自己才能触发
          <div class="box2" v-on:click="say2">
              <div class="box3" v-on:click="say3">谁弹出来</div>
          </div>
      </div>
  </div>

      <script>
      new Vue({
          el: "#app",
          methods: {
              say1() {
                  alert("box1")
              },
              say2() {
                  alert("box2")
              },
              say3() {
                  alert("box3")
              }
          }
      })
  </script>

c .prevent → 阻止默认行为，添加prevent之后就算之前设置了跳转链接，也是不会跳转的，如下：

    <div id="app">
      <a href="http://www.baidu.com" v-on:click.prevent="fn">百度</a>
  </div>
      <script>
      new Vue({
          el: "#app",
          methods: {
              fn() {
                  window.location.href = "http://jd.com"
              }
          }
  </script>

d .self → 当事件源等于自身的时候才触发，只有真正得点击自己自身盒子得时候，才会触发事件，如下：

  <div id="app">
      <div class="box1" v-on:click.self="say1">
          加了self以后只能点自己才能触发
          <div class="box2" v-on:click="say2">
              <div class="box3" v-on:click="say3">谁弹出来</div>
          </div>
      </div>
  </div>

      <script>
      new Vue({
          el: "#app",
          methods: {
              say1() {
                  alert("box1")
              },
              say2() {
                  alert("box2")
              },
              say3() {
                  alert("box3")
              }
          }
      })
  </script>

e .once → 点击只能触发一次 如下：

  <div id="app">
      <div class="box1" v-on:click.once="say1">
          加了self以后只能点自己才能触发
          <div class="box2" v-on:click="say2">
              <div class="box3" v-on:click="say3">谁弹出来</div>
          </div>
      </div>
  </div>

      <script>
      new Vue({
          el: "#app",
          methods: {
              say1() {
                  alert("box1")
              },
              say2() {
                  alert("box2")
              },
              say3() {
                  alert("box3")
              }
          }
      })
  </script>
